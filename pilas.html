                        pila.push("Casa");

                        pila.push("Radio");
                        pila.push("Audifonos");
                        pila.push("Foco");
                        
                        // Mostrar la pila
                        System.out.println("La pila es: " + pila);

                        // Mostrar el elemento en el tope de la pila con
                        // la función peek
                        System.out.println("El tope de la pila es: " + pila.peek());
                        
                        // Retirar elementos de la pila con 
                        // la función pop
                        System.out.println("Elemento retirado: " + pila.pop());
                                
                        for(int i = 0; i < 5; i++){
                            // Comprobar si la pila esta vacia

                                System.out.println("Elemento retirado con el for: " + pila.pop());
                                
                            } else {
                                System.out.println("No hay más elementos");
                                break;
                            }
                        }
                    }
                }
