                package com.mycompany.estructurasdedatos;

                import java.util.LinkedList;
                import java.util.Queue;
                /**
                *
                * @author tack
                */
                public class Colas {
                    public static void main(String[] args) {
                        
                        //Creamos la cola 
                        Queue cola = new LinkedList<>();
                
                        // Agregamos elementos a la cola con 
                        // la función add
                        cola.add("Manzana");
                        cola.add("Platano");
                        cola.add("Cereza");
                
                        // Mostrar la cola
                        System.out.println("La cola es: " + cola);
                
                        // Retirar el elemento que esta al frente de la cola con
                        // la función remove
                        System.out.println("Elemento retirado: " + cola.remove());
                
                        // Mostrar la cola despues de retirar el elemento
                        System.out.println("La cola actualizada es: " + cola);
                
                        // Mostrar el elemento al frente de la cola
                        System.out.println("El elemento al frente es: " + cola.peek());
                    }
                }
